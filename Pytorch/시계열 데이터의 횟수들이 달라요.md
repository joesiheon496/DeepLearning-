```python
import torch
from torch.nn.utils.rnn import pad_sequence

# 시계열 데이터와 해당 측정값 개수
time_series_data = [
    torch.tensor([1, 2, 3]),
    torch.tensor([4, 5]),
    torch.tensor([6, 7, 8, 9]),
    torch.tensor([10])
]
num_measurements = [3, 2, 4, 1]

# 패딩을 적용하여 시계열 데이터를 동일한 길이로 만들기
padded_data = pad_sequence(time_series_data, batch_first=True)

# 패딩 적용된 데이터 출력
print(padded_data)
```
